\def\printnum{
	\count6\count5
	\divide\count6 255
	\edef\tmp{\tmp\the\count6}
	\multiply\count6 -255
	\advance\count5 \count6
	\multiply\count5 10
}
\def\cset#1#2#3
{
	\uppercase{\count5="#2#3}\relax
	\def\tmp{}\printnum\edef\tmp{\tmp.}
	\printnum\printnum
	\printnum\printnum
	\expandafter\edef\csname #1\endcsname,{\tmp}
}
\def\cdef#1#2#3#4#5#6
{
	\cset{cr}#1#2
	\cset{cg}#3#4
	\cset{cb}#5#6
}
