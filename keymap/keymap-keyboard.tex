\expandafter \ifx \csname Key\endcsname \relax
\def\Key#1#2#3{}
\fi
\def\setKeyByName#1#2#3{%
\expandafter \ifx \csname setKeyName\endcsname \relax \else \setKeyName{#1}{#2}{#3}\fi
\ex\edef\csname Key#3Row\endcsname{#1}%
\ex\edef\csname Key#3Col\endcsname{#2}%
\edef\tmp##1{\noexpand\Key{##1}{#1}{#2}}%
\letcs{Key#3}{tmp}
}
\def\setKeyByNames#1#2#3{%
\countj#2%
\for{iter}\in{#3}{%
\setKeyByName{#1}{\the\countj}{\iter}%
\advance\countj1}%
}
